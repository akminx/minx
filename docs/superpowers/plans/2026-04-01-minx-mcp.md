**Goal:** Build a new portable `minx-mcp` repo that replaces the current SousChef and FinanceHub split with shared foundations, finance/health/meals/dashboard MCP surfaces, a separate dashboard HTTP app, and Hermes integration adapters.

**Architecture:** `minx-mcp` is a monorepo with shared foundation modules, shared transport bootstrapping, and separate finance, health, meals, and dashboard MCP entry points. The dashboard browser UI is a separate HTTP process in the same repo that reads the shared SQLite database directly and uses Hermes only as a chat backend.

**Tech Stack:** Python 3.12, SQLite, MCP Python SDK, FastAPI, Svelte, Chart.js, pytest, LiteParse, Obsidian skills, uv or venv-based packaging

- [ ] **Step 2: Implement `db.py`**

Use this structure:

```python
from __future__ import annotations

import sqlite3
from pathlib import Path

from minx_mcp.config import MINX_DB_PATH


_CONNECTIONS: dict[Path, sqlite3.Connection] = {}


def _migration_dir() -> Path:
    return Path(__file__).resolve().parent.parent / "schema" / "migrations"


def get_connection() -> sqlite3.Connection:
    db_path = MINX_DB_PATH
    db_path.parent.mkdir(parents=True, exist_ok=True)
    if db_path not in _CONNECTIONS:
        conn = sqlite3.connect(str(db_path), check_same_thread=False)
        conn.row_factory = sqlite3.Row
        conn.execute("PRAGMA journal_mode=WAL")
        conn.execute("PRAGMA foreign_keys=ON")
        _apply_migrations(conn)
        _CONNECTIONS[db_path] = conn
    return _CONNECTIONS[db_path]


def _apply_migrations(conn: sqlite3.Connection) -> None:
    conn.execute(
        """
        CREATE TABLE IF NOT EXISTS _migrations (
            name TEXT PRIMARY KEY,
            applied_at TEXT DEFAULT (datetime('now'))
        )
        """
    )
    applied = {
        row["name"]
        for row in conn.execute("SELECT name FROM _migrations").fetchall()
    }
    for path in sorted(_migration_dir().glob("*.sql")):
        if path.name in applied:
            continue
        conn.executescript(path.read_text())
        conn.execute("INSERT INTO _migrations (name) VALUES (?)", (path.name,))
    conn.commit()
```
